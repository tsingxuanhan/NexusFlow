### 3.4 检索结果长度控制策略

| 场景 | 最大Token | 策略 |
|------|-----------|------|
| 单次引用 | 500 tokens | 精确匹配片段 |
| 上下文注入 | 2000 tokens | 摘要+关键片段 |
| 深度分析 | 5000 tokens | 完整相关文档 |

```python
# 检索结果处理伪代码
def process_retrieval(query: str, max_tokens: int = 2000) -> str:
    results = chroma.query(query, top_k=5)
    
    if len(results) == 0:
        return ""  # 无结果，返回空
    
    # 按相似度排序
    sorted_results = sorted(results, key=lambda x: x.score, reverse=True)
    
    # 累积内容直到达到token限制
    accumulated = []
    current_tokens = 0
    
    for r in sorted_results:
        r_tokens = estimate_tokens(r.content)
        if current_tokens + r_tokens <= max_tokens:
            accumulated.append(f"[来源: {r.source}]\n{r.content}")
            current_tokens += r_tokens
        else:
            break  # 达到限制，停止
    
    return "\n\n---\n\n".join(accumulated)
```

async function checkService(service) {
  const endpoints = {
    ollama: '/api/tags',
    openwebui: '/api/v1/config',
    openclaw: '/health',
    portainer: '/api/system/status',
    comfyui: '/system_stats',
    jupyter: '/api',
    drawio: '/',
    stirling: '/api/v1/info',
  };
  
  const endpoint = endpoints[service];
  if (!endpoint) return { status: 'unknown', latency: 0 };
  
  const startTime = Date.now();
  try {
    const controller = new AbortController();
    const timeout = setTimeout(() => controller.abort(), 3000);
    
    const response = await fetch(API[service] + endpoint, {
      signal: controller.signal,
      method: 'GET',
    });
    
    clearTimeout(timeout);
    const latency = Date.now() - startTime;
    
    return {
      status: response.ok ? 'online' : 'error',
      latency: latency,
    };
  } catch (e) {
    return { status: 'offline', latency: 0, error: e.message };
  }
}

  const results = {};

}

// 格式化文件大小
function formatBytes(bytes) {
  if (bytes === 0) return '0 B';
  const k = 1024;
  const sizes = ['B', 'KB', 'MB', 'GB'];
  const i = Math.floor(Math.log(bytes) / Math.log(k));
  return parseFloat((bytes / Math.pow(k, i)).toFixed(1)) + ' ' + sizes[i];
}

// 格式化相对时间
function formatRelativeTime(timestamp) {
  const now = Date.now();
  const diff = now - timestamp;
  
  if (diff < 60000) return '刚刚';
  if (diff < 3600000) return Math.floor(diff / 60000) + '分钟前';
  if (diff < 86400000) return Math.floor(diff / 3600000) + '小时前';
  return new Date(timestamp).toLocaleDateString('zh-CN');
}

  const latencyText = status.latency ? `${status.latency}ms` : '-';

      </div>
      <button class="btn">
        ${status.status === 'online' ? '打开 →' : '启动'}
      </button>
    </div>
  `;
}

#!/usr/bin/env python3
"""RAG向量索引脚本"""
import chromadb
from chromadb.config import Settings
from pathlib import Path
import os

# 配置
WIKI_DIR = "D:\\AI\\XuanShu\\.wiki"
VECTOR_DIR = "D:\\AI\\XuanShu\\.vector-store"
COLLECTION = "xuan-hub-knowledge"

def init_chroma():
    client = chromadb.PersistentClient(path=VECTOR_DIR)
    collection = client.get_or_create_collection(
        name=COLLECTION,
        metadata={"description": "铉枢共享知识库"}
    )
    return collection

def index_directory(collection, dir_path):
    """索引目录下的所有.md文件"""
    for md_file in Path(dir_path).rglob("*.md"):
        # 读取文件
        with open(md_file, 'r', encoding='utf-8') as f:
            content = f.read()
        
        # 添加到向量库
        collection.add(
            documents=[content],
            ids=[str(md_file)],
            metadatas=[{
                "source": str(md_file),
                "type": "markdown",
                "modified": os.path.getmtime(md_file)
            }]
        )
        print(f"Indexed: {md_file}")

if __name__ == "__main__":
    collection = init_chroma()
    index_directory(collection, WIKI_DIR)
    print(f"Total documents: {collection.count()}")
EOF
